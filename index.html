// GET /api/whatsapp/messages?since= — o CRM puxa as mensagens recebidas (Supabase) import { sbSelect } from "../_supabase.js"; export default async function handler(req, res) { res.setHeader("Access-Control-Allow-Origin", "*"); const since = Number(req.query.since || 0); const rows = await sbSelect("messages", `t=gt.${since}&order=t.asc&limit=100`); return res.status(200).json({ messages: rows }); }
